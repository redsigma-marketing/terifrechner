<script>
document.getElementById('kundentyp').addEventListener('change', function() {
  document.getElementById('inputFields').style.display = 'block';

  if (this.value === 'privat') {
    document.getElementById('label1').textContent = 'Personen im Haushalt';
  } else if (this.value === 'kmu') {
    document.getElementById('label1').textContent = 'Mitarbeiterzahl';
  } else {
    document.getElementById('label1').textContent = 'Mitarbeiterzahl / Bereich';
  }
});

function berechnen() {
  let strom = parseFloat(document.getElementById('strom').value) || 0;
  let gas = parseFloat(document.getElementById('gas').value) || 0;

  // Durchschnittswerte (Oktober 2025 baseline)
  let stromNeu = 0.262; // 26,2 ct/kWh
  let gasNeu = 0.07;    // 7 ct/kWh

  let summe = (strom * stromNeu) + (gas * gasNeu);
  summe = summe.toFixed(2);

  document.getElementById('result').textContent =
    "Geschätztes Einsparpotenzial: " + summe + " € pro Jahr";

  document.getElementById('footnote').style.display = 'block';
}
</script>
